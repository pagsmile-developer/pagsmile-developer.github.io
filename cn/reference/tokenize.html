# 获取信用卡Token

# 请求接口ROOT地址

  测试环境 : https://security-test.pagsmile.com
  正式环境 : https://security.pagsmile.com

# 接口

  /card/token

# Request Header

参数名 必须 描述
Content-Type 推荐 application/json
Authorization yes Basic Base64(app_id:security_key)

# 请求参数（JSON格式）

参数名 类型 必须 最大长度/默认值 描述
app_id string yes 32 商户后台创建并获得
timestamp string yes 19 yyyy-MM-dd HH:mm:ss
card.card_no string yes 32 信用卡卡号
card.issuer string yes 16 卡组织代码(visa,mastercard...)
card.holder.name string yes 64 持卡人姓名
card.holder.identification.type string no 16 持卡人证件类型
card.holder.identification.number string yes 64 持卡人证件号
card.cvv string yes 8 安全码
card.valid_thru_year string yes 4 信用卡过期年
card.valid_thru_month string yes 2 信用卡过期月

# 请求样例

curl --location --request POST 'https://security.pagsmile.com/card/token' \
--header 'Authorization: Basic Base64(appid:security_key)' \
--header 'Content-Type: application/json' \
--data-raw '{
    "app_id": "app_id",
    "timestamp": "{{datetime}}",
    "card": {
      "card_no": "card_no",
      ...
    }
}'

# 响应（JSON格式）

参数名 类型 描述
code string 返回码
msg string 返回消息
sub_code string 子码
sub_msg string 子消息
token string

# 返回样例（成功）

{
    "code": "10000",
    "msg": "Success",
    "token": "{token}",
}

# 返回样例（失败）

{
    "code": "40002",
    "msg": "Business Failed",
    "sub_code": "invalid-signature",
    "sub_msg": "invalid signature"
}
