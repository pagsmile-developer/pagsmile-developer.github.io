# 直连接口下单并支付

# 请求接口ROOT地址

  测试环境 : https://gateway-test.pagsmile.com
  正式环境 : https://gateway.pagsmile.com

# 接口

  /trade/pay

# Request Header

参数名 必须 描述
Content-Type 推荐 application/json
Authorization yes Basic Base64(app_id:secret_key)

# 请求参数（JSON格式）

参数名 类型 必须 最大长度/默认值 描述
app_id string yes 32 商户后台创建并获得
timestamp string yes 19 yyyy-MM-dd HH:mm:ss
out_trade_no string yes 64 商户订单号
method string yes 32
order_currency string yes 3 巴西（BRL）
order_amount decimal yes 0.01 ~ 99999999999999.99 订单金额
subject string yes 128 订单标题
content string no 255 订单内容
notify_url string yes 回调地址
buyer_id string yes 商户的用户ID
timeout_express string no 90m m(分), h(小时), d(日), c(当日)
token string no 信用卡token（信用卡方式必须）
issuer string no 信用卡组织（信用卡方式必须）
installments integer no 1 信用卡分期
bank string no 银行（DepositExpress方式必须）
customer.name string yes 用户姓名
customer.email string yes 用户邮箱
customer.phone string yes 用户电话
customer.identify.number string yes 用户ID
customer.identify.type string yes 用户ID类型
address.zip_code string yes 邮编
address.state string yes 区
address.city string yes 城市
address.street_number string yes 街道1号码
address.street string yes 街道1
address.neighborhood string no 街道2
user_ip string no 用户IP（信用卡方式必传）

# 请求样例

curl --location --request POST 'https://gateway.pagsmile.com/trade/pay' \
--header 'Authorization: Basic Base64(appid:secret_key)' \
--header 'Content-Type: application/json' \
--data-raw '{
    "app_id": "app_id",
    "content": "content",
    "method": "Boleto",
    "notify_url": "notify_url",
    "order_amount": 10,
    "order_currency": "BRL",
    "out_trade_no": "{{$randomUUID}}",
    "subject": "subject",
    "timeout_express": "1h",
    "timestamp": "{{datetime}}",
    "customer": {
      "email": "email",
      ...
    },
    "address": {
      "zip_code": "zip_code",
      ...
    },
    "user_ip": "127.0.0.1"
}'

# 响应（JSON格式）

参数名 类型 描述
code string 返回码
msg string 返回消息
sub_code string 子码
sub_msg string 子消息
out_trade_no string 商户订单号
trade_no string Pagsmile创建的订单号
trade_status string Pagsmile创建的订单状态

# 返回样例（成功）

{
    "code": "10000",
    "msg": "Success",
    "out_trade_no": "{out_trade_no}",
    "trade_no": "{trade_no}",
    "trade_status": "{trade_status}"
}

# 返回样例（失败）

{
    "code": "40002",
    "msg": "Business Failed",
    "sub_code": "invalid-signature",
    "sub_msg": "invalid signature"
}
