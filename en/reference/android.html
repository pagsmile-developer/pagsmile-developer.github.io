# SDK Android

# 1. Add aar

Import the following aar package into libs under the application Android project.
pagsmile-1.0.0.aar
livechat.aar
react-native-0.61.2.aar
react-native-i18n.aar
react-native-pdf.aar
rn-fetch-blob.aar
react-native-vector-icons.aar
hermes-debug.aar
hermes-release.aar

# 2. Add Configuration in the Project's build.gradle File:

// Top-level build file where you can add configration options common to all sub-projects/modules.

buildscript {
  repositories {
    mavenCentral()
    ...
  } ...
}
allprojects {
  repositories {
    ...
  }
}
mavenCentral()
maven { url 'https://jitpack.io' }

# 3. Add Configuration in App's build.gradle:

android {
  ...
  defaultConfig {
    multiDexEnabled true
    ndk {
      abiFilters "armeabi-v7a", "x86"
    }
  }
  packagingOptions {
      exclude "lib/arm64-v8a/librealm-jni.so"
      pickFirst 'lib/x86/libc++_shared.so'
      pickFirst 'lib/arm64-v8a/libc++_shared.so'
      pickFirst 'lib/armeabi-v7a/libc++_shared.so'
      pickFirst 'lib/x86_64/libc++_shared.so'
  }
  repositories {
      flatDir {
          dirs 'libs'
      }
  }
  ...
}

dependencies {
  implementation fileTree(dir: 'libs', include: ['*.jar'])

  implementation 'androidx.appcompat:appcompat:1.1.0'
  implementation 'com.android.support:multidex:2.0.3'

  implementation(name: 'livechat', ext: 'aar')

  implementation(name: 'react-native-pdf', ext: 'aar')

# 4. AndroidManifest.xml

If you don’t have your own application file, add attributes to the application tag of the AndroidManifest.xml file

android:name="com.pagsmile.sdk.PagsmileApplication"

If you have your own application file, add the following code:

@Override
protected void attachBaseContext(Context base) {
  super.attachBaseContext(base);
  MultiDex.install(this);
}

# 5. Initialize SDK

It is recommended to execute it when the application starts or after successful login

Pagsmile.getInstance().init(Context, app_id, api_key, Pagsmile.Test);

# 6. Pull Up the Order Page

Call it after getting the prepay_id returned by pagsmile background

Pagsmile.getInstance().order(Context, prepayId);
